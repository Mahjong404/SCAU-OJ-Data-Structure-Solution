#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

int main()
{
    int n;
    cin>>n;
    vector<int> a(n);
    for(int i=0;i<n;i++)
        cin>>a[i];
    for(int i=1;i<n;i++){
        sort(a.begin(),a.begin()+i+1);
        for(int j=0;j<n;j++)
            cout<<a[j]<<" ";
        cout<<endl;
    }
    return 0;
}
